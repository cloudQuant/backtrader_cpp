cmake_minimum_required(VERSION 3.16)
project(BacktraderSMATest)

# 设置C++标准
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 查找Google Test
find_package(GTest REQUIRED)
find_package(Threads REQUIRED)

# 包含目录
include_directories(
    ${CMAKE_SOURCE_DIR}/../../include
    ${CMAKE_SOURCE_DIR}/..
    ${GTEST_INCLUDE_DIRS}
)

# 获取SMA测试必需的最小源文件
set(BACKTRADER_SRC_DIR ${CMAKE_SOURCE_DIR}/../../src)

# 只包含SMA测试需要的核心文件
set(BACKTRADER_MINIMAL_SOURCES
    "${BACKTRADER_SRC_DIR}/lineroot.cpp"
    "${BACKTRADER_SRC_DIR}/linebuffer.cpp"
    "${BACKTRADER_SRC_DIR}/lineseries.cpp"
    "${BACKTRADER_SRC_DIR}/lineiterator.cpp"
    "${BACKTRADER_SRC_DIR}/dataseries.cpp"
    "${BACKTRADER_SRC_DIR}/indicator.cpp"
    "${BACKTRADER_SRC_DIR}/indicators/sma.cpp"
)

# 创建最小的backtrader库
add_library(backtrader_minimal_core STATIC ${BACKTRADER_MINIMAL_SOURCES})
target_link_libraries(backtrader_minimal_core Threads::Threads)

# 创建SMA测试可执行文件
add_executable(test_ind_sma_minimal test_ind_sma.cpp)
target_link_libraries(test_ind_sma_minimal
    backtrader_minimal_core
    ${GTEST_LIBRARIES}
    gtest_main
    Threads::Threads
)

add_test(NAME test_ind_sma_minimal COMMAND test_ind_sma_minimal)

# 打印配置信息
message(STATUS "=== Minimal SMA Test Configuration ===")
message(STATUS "Google Test: ${GTEST_FOUND}")
message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")
message(STATUS "=======================================")
