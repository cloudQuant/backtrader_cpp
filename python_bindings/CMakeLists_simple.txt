cmake_minimum_required(VERSION 3.16)
project(backtrader_cpp_simple_bindings VERSION 0.1.0)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find required packages
find_package(pybind11 REQUIRED HINTS ${PYBIND11_ROOT})
find_package(Python3 COMPONENTS Interpreter Development REQUIRED)

# Create Python module - simplified version
pybind11_add_module(backtrader_cpp src/main_simple.cpp)

# Compiler-specific options
target_compile_definitions(backtrader_cpp PRIVATE 
    VERSION_INFO="${PROJECT_VERSION}"
    PYBIND11_DETAILED_ERROR_MESSAGES
)

# Set properties
set_target_properties(backtrader_cpp PROPERTIES
    CXX_VISIBILITY_PRESET "hidden"
    VISIBILITY_INLINES_HIDDEN YES
)

# Optimization flags for release builds
target_compile_options(backtrader_cpp PRIVATE
    $<$<CONFIG:Release>:-O3 -march=native -DNDEBUG>
    $<$<CONFIG:Debug>:-g -O0 -DDEBUG>
)

# Install configuration
install(TARGETS backtrader_cpp DESTINATION .)

# Print configuration
message(STATUS "=== Backtrader C++ Simple Bindings Configuration ===")
message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")
message(STATUS "C++ Standard: ${CMAKE_CXX_STANDARD}")
message(STATUS "Python: ${Python3_VERSION}")
message(STATUS "Pybind11: ${pybind11_VERSION}")
message(STATUS "======================================================")
