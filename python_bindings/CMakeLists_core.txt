cmake_minimum_required(VERSION 3.16)
project(backtrader_cpp_core_bindings VERSION 0.2.0)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find required packages
find_package(pybind11 REQUIRED HINTS ${PYBIND11_ROOT})

# Include directories for backtrader-cpp
include_directories(${CMAKE_CURRENT_SOURCE_DIR}/../../include)

# Link directories for core library
link_directories(${CMAKE_CURRENT_SOURCE_DIR}/../../build)
link_directories(${CMAKE_CURRENT_SOURCE_DIR}/../..)

# Create Python module with core library integration
pybind11_add_module(backtrader_cpp ../src/main_core.cpp)

# Link with backtrader core library
target_link_libraries(backtrader_cpp PRIVATE backtrader_core)

# Compiler-specific options
target_compile_definitions(backtrader_cpp PRIVATE 
    VERSION_INFO="${PROJECT_VERSION}"
)

# Set properties for better compatibility
set_target_properties(backtrader_cpp PROPERTIES
    CXX_VISIBILITY_PRESET "hidden"
    VISIBILITY_INLINES_HIDDEN YES
    LINK_FLAGS "-static-libgcc -static-libstdc++"
)

# Optimization flags
target_compile_options(backtrader_cpp PRIVATE
    $<$<CONFIG:Release>:-O3 -march=native -DNDEBUG>
    $<$<CONFIG:Debug>:-g -O0 -DDEBUG>
)

message(STATUS "=== Backtrader C++ Core Bindings ===")
message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")
message(STATUS "C++ Standard: ${CMAKE_CXX_STANDARD}")
message(STATUS "Core library: ${CMAKE_CURRENT_SOURCE_DIR}/../../libbacktrader_core.a")
message(STATUS "Include path: ${CMAKE_CURRENT_SOURCE_DIR}/../../include")
message(STATUS "======================================")
