cmake_minimum_required(VERSION 3.16)
project(backtrader_cpp_working_bindings VERSION 0.3.0)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find required packages
find_package(pybind11 REQUIRED HINTS ${PYBIND11_ROOT})

# Create Python module - working version with comprehensive functionality
pybind11_add_module(backtrader_cpp ../src/main_working.cpp)

# Compiler-specific options
target_compile_definitions(backtrader_cpp PRIVATE 
    VERSION_INFO="${PROJECT_VERSION}"
)

# Set properties for better compatibility
set_target_properties(backtrader_cpp PROPERTIES
    CXX_VISIBILITY_PRESET "hidden"
    VISIBILITY_INLINES_HIDDEN YES
    LINK_FLAGS "-static-libgcc -static-libstdc++"
)

# Optimization flags
target_compile_options(backtrader_cpp PRIVATE
    $<$<CONFIG:Release>:-O3 -march=native -DNDEBUG>
    $<$<CONFIG:Debug>:-g -O0 -DDEBUG>
)

message(STATUS "=== Backtrader C++ Working Bindings ===")
message(STATUS "Build type: ${CMAKE_BUILD_TYPE}")
message(STATUS "C++ Standard: ${CMAKE_CXX_STANDARD}")
message(STATUS "Features: Math Functions, Strategy Simulation, Performance Testing")
message(STATUS "========================================")
